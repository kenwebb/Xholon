package org.primordion.user.app.climatechange.carboncycle03;

import org.primordion.xholon.base.IXholon;

import com.kitfox.svg.Text;
import com.kitfox.svg.Tspan;

public class SvgLabel extends Xhcarboncycle03 {
	
	private String svgId = "";
	private String svgLabel = "";
	
	private IXholon carbonCycleView = null;
	
	public void postConfigure() {
		setText();
		super.postConfigure();
		this.removeChild();
	}
	
	/**
	 * Set the text for an existing SVG tspan node, whose parent must be an SVG text node.
	 */
	protected void setText() {
		Tspan svgNode = (Tspan)((CarbonCycleView)carbonCycleView).getDiagram().getElement(svgId);
		if (svgNode == null) {return;}
		svgNode.setText(svgLabel);
        Text textEle = (Text)svgNode.getParent();
        try {
            textEle.rebuild();
        } catch (Exception e) {
            e.printStackTrace();
        }
	}
	
	/*
	 * @see org.primordion.xholon.base.Xholon#setAttributeVal(java.lang.String, java.lang.String)
	 */
	public int setAttributeVal(String attrName, String attrVal) {
		if ("svgId".equals(attrName)) {
			svgId = attrVal;
		}
		else if ("svgLabel".equals(attrName)) {
			svgLabel = attrVal;
		}
		return 0;
	}
	
	public String toString() {
		String outStr = getName();
		outStr += " svgId:" + svgId;
		outStr += " svgLabel:" + svgLabel;
		return outStr;
	}

	public String getSvgId() {
		return svgId;
	}

	public void setSvgId(String svgId) {
		this.svgId = svgId;
	}

	public String getSvgLabel() {
		return svgLabel;
	}

	public void setSvgLabel(String svgLabel) {
		this.svgLabel = svgLabel;
	}

	public IXholon getCarbonCycleView() {
		return carbonCycleView;
	}

	public void setCarbonCycleView(IXholon carbonCycleView) {
		this.carbonCycleView = carbonCycleView;
	}
	
}
